{{- /* Mô tả danh sách: ưu tiên description, không có thì rút từ Summary hoặc Plain. Dùng replaceRE thay strings.TrimSpace — Hugo 0.92 (Ubuntu) chưa có strings.TrimSpace. */ -}} {{- $d := "" -}} {{- if .Description -}} {{- $d = .Description | replaceRE "^\\s+|\\s+$" "" -}} {{- else if .Summary -}} {{- $d = .Summary | plainify | replaceRE "^\\s+|\\s+$" "" -}} {{- else -}} {{- $d = .Plain | plainify | truncate 200 | replaceRE "^\\s+|\\s+$" "" -}} {{- end -}} {{- with $d -}} {{ . }} {{- end -}}
